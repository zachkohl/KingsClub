https://docs.aws.amazon.com/AmazonS3/latest/dev/example-bucket-policies.html

Update user permision by directly editing JSON.
The bucket must reference the user, and the user must reference the bucket.

<CORSConfiguration xmlns="http://s3.amazonaws.com/doc/2006-03-01/">

    <AllowedOrigin>https://kings-club.herokuapp.com</AllowedOrigin>
    <AllowedOrigin>http://AE9E5CF3EEFF2DCA</AllowedOrigin>
    <AllowedMethod>GET</AllowedMethod>
    <AllowedMethod>POST</AllowedMethod>
    <AllowedMethod>PUT</AllowedMethod>
    <AllowedHeader>*</AllowedHeader>
</CORSRule>
</CORSConfiguration>


Example Bucket Policy:
{
    "Version": "2012-10-17",
    "Statement": [
        {
            "Sid": "AddCannedAcl",
            "Effect": "Allow",
            "Principal": {
                "AWS": [
                    "arn:aws:iam::803875257005:user/kings-club"
                ]
            },
            "Action": "*",
            "Resource": [
                "arn:aws:s3:::kings-club/*"
            ],
            "Condition": {
                "StringEquals": {
                    "s3:x-amz-acl": [
                        "public-read"
                    ]
                }
            }
        }
    ]
}

Example User Inline Policy:
{
    "Version": "2012-10-17",
    "Statement": [
        {
            "Effect": "Allow",
            "Action": [
                "s3:ListAllMyBuckets"
            ],
            "Resource": "arn:aws:s3:::*"
        },
        {
            "Effect": "Allow",
            "Action": [
                "s3:ListBucket",
                "s3:GetBucketLocation"
            ],
            "Resource": "arn:aws:s3:::kings-club"
        },
        {
            "Effect": "Allow",
            "Action": [
                "s3:PutObject",
                "s3:PutObjectAcl",
                "s3:GetObject",
                "s3:GetObjectAcl",
                "s3:DeleteObject"
            ],
            "Resource": "arn:aws:s3:::kings-club/*"
        }
    ]
}
